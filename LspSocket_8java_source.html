117  final LspConnection connect(SocketAddress sockAddr, LspParams params, ConnectionTriggers triggers) throws TimeoutException {
118  synchronized (connectExecutor) {
119  // Inicia uma nova tarefa de conexão
120  connTask = new ConnectTask(sockAddr);
121  Future<Short> call = connectExecutor.submit(connTask);
122 
123  try {
124  int limit = params.getEpochLimit();
125  while (limit > 0) {
126  try {
127  short connId = call.get(params.getEpoch(), TimeUnit.MILLISECONDS);
128  return new LspConnection(connId, sockAddr, params, triggers);
129  } catch (TimeoutException e) {
130  --limit;
131  } catch (InterruptedException e) {
132  return null;
133  } catch (ExecutionException e) {
134  throw new RuntimeException(e);
135  }
136  }
137  } finally {
138  connTask.ack((short) 0); // libera a thread
139  connTask = null;
140  }
141 
142  throw new TimeoutException("Servidor " + sockAddr + " não responde");
143  }

180  void dgramReceiveData(final SocketAddress sockAddr, final ByteBuffer buf) {
181  LspConnection conn = usedConnection(sockAddr, buf.getShort());
182 
183  // Só continua se a conexão é válida e não estiver fechada
184  if (conn != null && !conn.isClosed()) {
185  short seqNum = buf.getShort();
186  byte[] payload = payload(buf);
187  InternalPack pack = new InternalPack(conn, seqNum, payload);
188 
189  // Se a mensagem foi enfileirada, envia o ACK e informa o número
190  // de sequência à conexão (usado nos disparos da época).
191  if (inputQueue.offer(pack)) {
192  dgramSendAck(pack);
193  conn.received(seqNum);
194  }
195 
196  // Caso contrário, mesmo que a mensagem não possa ser lida,
197  // atualiza o momento da última mensagem recebida
198  else {
199  conn.received();
200  }
201  }

205  void dgramReceiveAck(final SocketAddress sockAddr, final ByteBuffer buf) {
206  final short connId = buf.getShort();
207  final LspConnection conn = usedConnection(sockAddr, connId);
208 
209  // Se o connId é válido, reconhece a mensagem
210  if (conn != null) {
211  final short seqNum = buf.getShort();
212  conn.ack(seqNum);
213  }
214 
215  // Senão verifica se há uma tentativa de conexão em curso. Caso
216  // positivo, verifica também se id não é 0, número de sequência é 0,
217  // conferindo antes se o ACK vem do socket remoto correto
218  else if (connTask != null && connId != 0 && buf.getShort() == 0
219  && sockAddr.equals(connTask.sockAddr)) {
220  connTask.ack(connId);
221  }

374  private void sendNextData() throws InterruptedException {
375  // Obtém o próximo pacote de dados da fila, se houver.
376  final Pack p = outputQueue.poll(1, TimeUnit.SECONDS);
377  if (p == null) {
378  return;

382  LspConnection conn = usedConnection(p.getConnId());

395  // espera de um ACK) então devolve o pacote à fila (segunda posição)

398  outputQueue.offerFirst(p);
